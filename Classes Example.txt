class Worker
{
protected:
	float salary;
	float money;
	string name;
	string company;
public:
	Worker()
	{
		salary = 100;
		money = 10;
		name = "";
		company = "";
	}
	Worker(int sal, int mon, string nam, string com)
	{
		salary = sal;
		money = mon;
		name = nam;
		company = com;
	}
	float Getsalary()
	{
		return salary;
	}
	float Getmoney()
	{
		return money;
	}
	string Getname()
	{
		return name;
	}
	string Getcompany()
	{
		return company;
	}

	void PayDay()
	{
		money += salary;
	}

	void TooLowSal()
	{
		salary += 1000;
	}

	void Show()
	{
		cout << "  ---Profile---\n";
		cout << "    Name: " << Getname() << endl << "    Company: " << Getcompany() << endl << "    Salary: " << Getsalary() << endl << "    Total Money: " << Getmoney() << endl << endl;
	}
};
