#include <string>
#include <vector>

	exit(0);
}

void main()
{
	setlocale(LC_ALL, "Rus");

	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);

	string str;


	cout << "¬ведите строку:" << endl;
	getline(cin, str);

	for (int i = 0; i < str.length(); i++)
		if (str[i] == ' ' && str[i + 1] == ' ')
		{
			str.erase(str.begin() + i + 1);
			i = -1;
		}

	cout << str << endl;

	for (int i = 0; i < str.length(); i++)
		if (str[i] == '.' || str[i] == ',')
		{
			if (str.length() > i + 1)
				if (str[i + 1] == ' ')
				{
					str.erase(str.begin() + i + 1);
					i = -1;
					continue;
				}

			if (str.length() > 1 && i > 0)
				if (str[i - 1] == ' ')
				{
					str.erase(str.begin() + i - 1);
					i = -1;
					continue;
				}
		}

	cout << endl << str << endl;

	for (int i = 0; i < str.length(); i++)
	{
		if (str[i] == '.' || str[i] == ',')
			if (str.length() > i + 1)
				if (str[i] == str[i + 1])
				{
					str.erase(str.begin() + i + 1);
					i = -1;
				}
	}

	cout << endl << str << endl;

	for (int i = 0; i < str.length(); i++)
	{
		if (str[i] == ',' || str[i] == '.')
			if (str.length() > i + 1)
				if (str[i + 1] != ' ')
					str.insert(str.begin() + i + 1, ' ');
	}

	cout << endl << str << endl;

	system("pause");
}
