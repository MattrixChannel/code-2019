#include <iostream>
#include <string>

using namespace std;

void main()
{
	setlocale(LC_ALL, "Rus");

	string str;
	string word;

	cout << "Введите строку:" << endl;
	getline(cin, str);

	cout << "\nВведите слово:" << endl;

	getline(cin, word);
	int match = 0, match_c = 0;

	for (int i = 0; i < str.length(); i++)
	{
		for (int j = 0; j < word.length() && j + i < str.length() && j < str.length(); j++)
		{
			if (str[i + j] == word[j])
				match++;
			else
				match = 0;
		}
		if (match == word.length())
			match_c++;
	}

	if (match_c > 0)
		cout << "\nЕсть" << endl;
	else
		cout << "\nнет" << endl;
	system("pause");
}
