#include <iostream>
#include <ctime>

using namespace std;

int main()
{
	srand(time(NULL));
	setlocale(LC_ALL, "Rus");

	const int N = 6;

	int Mas[N], MEn[10], Num[6];

	MEn[0] = 6;
	MEn[1] = 2;
	MEn[2] = 5;
	MEn[3] = 5;
	MEn[4] = 4;
	MEn[5] = 5;
	MEn[6] = 6;
	MEn[7] = 3;
	MEn[8] = 7;
	MEn[9] = 6;

	for (int i = 0; i < N; i++)
	{
		Mas[i] = 0;
		Num[i] = 0;
	}
	Mas[5]--;
	while (true)
	{
		Mas[5]++;
		if (MEn[Num[5]] > MEn[Mas[5]])
			Num[5] = Mas[5];
		if (Mas[5] > 9)
		{
			Mas[5] = 0;
			if (MEn[Num[4]] > MEn[Mas[4]])
				Num[0] = Mas[4];
			Mas[4]++;
		}
		if (Mas[4] > 5)
		{
			Mas[4] = 0;
			if (MEn[Num[3]] > MEn[Mas[3]])
				Num[3] = Mas[3];
			Mas[3]++;
		}
		if (Mas[3] > 9)
		{
			Mas[3] = 0;
			if (MEn[Num[2]] > MEn[Mas[2]])
				Num[2] = Mas[2];
			Mas[2]++;
		}
		if (Mas[2] > 5)
		{
			Mas[2] = 0;
			if (MEn[Num[1]] < MEn[Mas[1]])
				Num[1] = Mas[1];
			Mas[1]++;
		}
		if (Mas[1] > 9)
		{
			Mas[1] = 0;
			if (MEn[Num[0]] < MEn[Mas[0]])
				Num[0] = Mas[0];
			Mas[0]++;
		}
		if (Mas[0] == 2 && Mas[1] == 4)
		{
			break;
		}
	}

	cout << Num[0] << Num[1] << ":" << Num[2] << Num[3] << ":" << Num[4] << Num[5] << endl;
	system("pause");
}
