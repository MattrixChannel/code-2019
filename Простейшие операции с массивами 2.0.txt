5. В одномерном массиве, состоящем из N элементов,
вычислить:
• минимальный элемент массива;
• сумму элементов массива, расположенных между первым и
последним положительными элементами.
Преобразовать массив таким образом, чтобы сначала располагались все
элементы, равные нулю, а потом все остальные.

#include <iostream>
#include <ctime>

using namespace std;

void randMas(int* m, int size)
{
	for (int i = 0; i < size; i++)
	{
		m[i] = rand() % 10 - 5;
	}
	cout << endl;
}

void prMas(int* m, int size)
{
	for (int i = 0; i < size; i++)
	{
		cout << m[i] << " ";
	}
	cout << endl;
}


int main()
{
	srand(time(NULL));
	setlocale(LC_ALL, "Rus");

	const int N = 10;

	int Mas[N], min = 0, fpel = 0, lpel = N - 1, sumel = 0;

	randMas(Mas, N);
	
	for (int i = 0; i < N; i++)
	{
		if (Mas[min] > Mas[i])
			min = i;
		if (Mas[fpel] <= 0)
			fpel++;
	}

	for (int i = N - 1; i >= 0; i--)
	{
		if (Mas[lpel] <= 0)
			lpel--;
	}

	for (int i = fpel + 1; i < lpel; i++)
	{
		sumel += Mas[i];
	}



	cout << "Массив:" << endl;

	prMas(Mas, N);

	cout << "\n Минимальный элемент массива -> " << Mas[min] << endl << endl;

	cout << " Сумма эл между первым и последним полох эл -> " << sumel << endl << endl;

	for (int i = 0; i < N; i++)
	{
		if (Mas[i] == 0)
			Mas[i] = -99;
	}

	for (int i = N - 1; i >= 0; i--)
	{
		if (Mas[i] == -99)
		{
			Mas[i] = 0;
			for (int j = i; j > 0; j--)
			{
				swap(Mas[j], Mas[j - 1]);
			}
		}
	}

	cout << endl;
	prMas(Mas, N);

	system("pause");
}
